CREATE TABLE Etudiants (
  id     INT        PRIMARY KEY

  age    INT        DEFAULT 0,

PDO — نظري فقط
prepare() • execute() • الحماية من SQL Injection • فهم الكود

🔄 كيف يعمل PDO — Flowchart

① الاتصال بقاعدة البيانات
↓
② prepare() — تحضير الاستعلام
↓
③ execute() — تنفيذ بالبيانات الآمنة
↓
④ fetch() / fetchAll() — جلب النتائج
// ① الاتصال
$pdo = new PDO(
  "mysql:host=localhost;dbname=mydb",
  "root",
  ""
);

// ② prepare — بدون بيانات حقيقية
$stmt = $pdo->prepare(
  "SELECT * FROM users 
   WHERE email = :email"
);

// ③ execute — البيانات تُدخَل بأمان
$stmt->execute([
    ':email' => $email
]);

// ④ fetchAll — كل النتائج
$users = $stmt->fetchAll();

📖 شرح كل دالة PDO

الدالةالدورلماذا هي مهمة؟
prepare() تحضير الاستعلام بدون بيانات حقيقية يفصل الكود SQL عن البيانات — أساس الأمان
execute() تنفيذ الاستعلام مع البيانات البيانات لا تُعاملَ كـ SQL أبداً — آمنة تماماً
:email / :nom Placeholder — مكان مؤقت للبيانات يمنع حقن SQL بشكل كامل
fetchAll() جلب كل النتائج كـ Array نستخدمها مع SELECT الذي يعطي أكثر من صف
fetch() جلب صف واحد فقط نستخدمها عند توقع نتيجة واحدة

❌ بدون PDO — خطر!

// بدون حماية — خطير جداً!
$user = $_POST['username'];
$sql  = "SELECT * FROM users 
  WHERE username='$user'";

// المهاجم يكتب في username:
// ' OR '1'='1
// الاستعلام يصبح:
// WHERE username='' OR '1'='1'
// → يتجاوز كلمة المرور!

✅ مع PDO — آمن!

$stmt = $pdo->prepare(
  "SELECT * FROM users
   WHERE username = :u
   AND password = :p"
);
$stmt->execute([
    ':u' => $_POST['username'],
    ':p' => $_POST['password']
]);
// حتى لو كتب ' OR '1'='1
// يُعامَل كنص عادي — لا SQL!

Sécurité — الأمان
الهجمات الرئيسية + وسائل الحماية — أهم نقطة في الامتحان
🔥 قال الأستاذ: هناك ~8 هجمات. يجب فهم وظيفة كل هجوم. الامتحان قد يعطيك كود ويسأل: "ما نوع الهجوم؟ كيف تحمي؟"

⚡ 1. SQL Injection — حقن SQL

الهدف: الدخول لقاعدة البيانات بدون إذن، قراءة/تعديل/حذف البيانات

كيف يعمل: يُدخل المهاجم كود SQL في حقل النموذج

مثال الهجوم:
Username: ' OR '1'='1 Password: anything

الحماية: PDO مع prepare() و execute()

⚡ 2. XSS — Cross-Site Scripting

الهدف: تنفيذ JavaScript في متصفح الضحية

كيف يعمل: يُدخل المهاجم كود HTML/JS في حقل نص

مثال الهجوم:
<script>alert('hacked!')</script>

❌ بدون حماية:

echo $_POST['comment'];
// ينفذ الـ script!

✅ مع الحماية:

echo htmlspecialchars(
  $_POST['comment'],
  ENT_QUOTES, 'UTF-8'
);
// <script> → &lt;script&gt;
// يظهر كنص ✅

📋 جدول الهجمات الـ 8 — ملخص للحفظ

#الهجومالهدفكيف يعملالحماية
1SQL Injection اختراق قاعدة البيانات حقن SQL في حقل نص PDO prepare()/execute()
2XSS تنفيذ JS في متصفح الضحية حقن <script> في حقل نص htmlspecialchars()
3CSRF تنفيذ طلبات باسم المستخدم رابط مزيف يُرسل طلب POST Token CSRF في النموذج
4Brute Force تخمين كلمة المرور تجربة آلاف الكلمات تلقائياً تحديد المحاولات + CAPTCHA
5Session Hijacking سرقة جلسة المستخدم سرقة session ID HTTPS + تجديد session_id()
6File Inclusion تضمين ملف خارجي خطير تغيير متغير include التحقق من اسم الملف
7Directory Traversal الوصول لملفات خارج المسموح ../../../etc/passwd basename() + realpath()
8Clickjacking خداع المستخدم بالنقر iframe شفاف فوق المحتوى X-Frame-Options header

🧹 دالة clean() — الحماية الشاملة من XSS

<?php
function clean($data) {
    return htmlspecialchars(
        $data,
        ENT_QUOTES,
        'UTF-8'
    );
}

// استخدام الدالة
$nom  = clean($_POST['nom']);
$msg  = clean($_POST['message']);
echo $nom; // آمن للعرض
?>
المعاملالوظيفة
htmlspecialchars()تحول < > & ' " إلى entities آمنة
ENT_QUOTESتحمي الاقتباسات المفردة والمزدوجة
'UTF-8'ترميز آمن لتجنب مشاكل التشفير
✅ نستدعي clean() على كل بيانات المستخدم قبل عرضها في الصفحة

⚡ PHP — تمارين TP

PHP — تمرين 1 | واش يظهر؟
هذا الكود واش يدير؟ وواش يطبع؟ <?php $a = 5; $b = 3; echo $a + $b . " هو المجموع"; ?>
▶ OUTPUT
8 هو المجموع
📖 PHP يحسب $a + $b = 8 أولاً، ثم يربط النتيجة بالنص بالنقطة .
PHP — تمرين 2 | if/else
ماذا يطبع هذا الكود؟ <?php $note = 9; if ($note >= 10) { echo "Admis"; } elseif ($note >= 8) { echo "Rattrapage"; } else { echo "Ajourné"; } ?>
▶ OUTPUT
Rattrapage
📖 9 ليس >= 10 → الشرط الأول خاطئ. 9 >= 8 → الشرط الثاني صحيح → يطبع "Rattrapage"
PHP — تمرين 3 | بoucle for
ماذا يطبع هذا الكود؟ <?php for ($i = 1; $i <= 5; $i++) { if ($i % 2 ! = 0) { echo $i . " "; } } ?>
▶ OUTPUT
1 3 5
📖 $i % 2 ! = 0 يعني "الباقي من القسمة على 2 ليس صفر" = الأرقام الفردية فقط (1, 3, 5)
PHP — تمرين 4 | foreach + Array
هذا الكود واش يدير؟ <?php $fruits = ["pomme", "orange", "banane"]; foreach ($fruits as $key => $val) { echo $key . " : " . $val . "<br>"; } ?>
▶ OUTPUT
0 : pomme 1 : orange 2 : banane
📖 foreach مع $key => $val يعطي الـ index (0,1,2) والقيمة في نفس الوقت
PHP — تمرين 5 | اكتشاف خطأ
أين الخطأ في هذا الكود؟ <?php $age = 20 echo "عمرك: " . $age ?>
🚨 خطأان:
① $age = 20 → ينقصها ; في النهاية
② echo "..." . $age → ينقصها ; في النهاية

القاعدة: كل سطر في PHP ينتهي بـ ; إجبارياً

🎨 CSS — تمارين TP

CSS — تمرين 1 | هذا الكود واش يدير؟
* { margin: 0; padding: 0; box-sizing: border-box; }
📖 هذا CSS Reset: يزيل كل المسافات الداخلية والخارجية الافتراضية للمتصفح من جميع العناصر. box-sizing: border-box يجعل الـ padding يُحسب داخل العرض وليس خارجه. يُستخدم في بداية كل stylesheet.
CSS — تمرين 2 | فهم السيليكتور
هذا الكود واش يدير؟ article:has(img) h2 + p:not(.error) { color: blue; font-weight: bold; }
📖 يستهدف: p الذي يجمع الشروط التالية:
① يأتي مباشرة بعد h2 (adjacent sibling: +)
② داخل article يحتوي على img (selector :has)
③ لا يحمل الـ class .error (selector :not)
→ النتيجة: النص يصبح أزرق وعريض
CSS — تمرين 3 | الفرق بين display:none و visibility:hidden
ما الفرق العملي بين display: none و visibility: hidden؟
display: nonevisibility: hidden
يُخفي العنصر؟نعمنعم
يشغل مساحة؟لا — يُزال من التدفقنعم — يبقى فراغ

CSS — تمرين 4 | اكتب الكود
اكتب CSS يجعل كل <h1> باللون الأحمر، توسيط، وحجم خط 2rem
h1 {
  color: red;

CSS — تمرين 5 | Sélecteur Descendant vs Child
ما الفرق بين div p و div > p؟
السيليكتورالمعنىمثال
div p كل p داخل div في أي مستوى (Descendant) يستهدف p داخل div حتى لو في span داخل div
div > p فقط p الابن المباشر لـ div (Child) لا يستهدف p داخل span داخل div

🗄️ SQL — تمارين TP

SQL — تمرين 1 | CREATE TABLE
اكتب SQL لإنشاء جدول Produits بـ: id (مفتاح أساسي تلقائي)، nom (نص إجباري)، prix (عشري)
CREATE TABLE Produits (
  id    INT          PRIMARY KEY AUTO_INCREMENT,
  nom   VARCHAR(100) NOT NULL,
  prix  DECIMAL(10,2)
);
SQL — تمرين 2 | SELECT مع WHERE
اكتب استعلام لعرض كل الطلاب الذين نقطتهم أكبر من أو تساوي 10، مرتبين تنازلياً حسب النقطة
SELECT * FROM Etudiants
WHERE note >= 10
ORDER BY note DESC;
SQL — تمرين 3 | UPDATE
اكتب SQL لتعديل نقطة الطالب ذو id=5 إلى 18
UPDATE Etudiants
SET note = 18
WHERE id = 5;
⚠️ WHERE إجبارية! بدونها يعدل كل الطلاب!
SQL — تمرين 4 | INSERT
أضف طالباً جديداً: الاسم "Sara"، النقطة 14.5، العمر 21
INSERT INTO Etudiants (nom, note, age)
VALUES ('Sara', 14.5, 21);
📌 id لا نكتبه — يزيد تلقائياً بسبب AUTO_INCREMENT
SQL — تمرين 5 | DELETE
احذف المنتج ذو id = 7
DELETE FROM Produits
WHERE id = 7;
🚨 WHERE إجبارية! بدونها يحذف كل الجدول!

🔐 PDO — تمارين نظرية

PDO — تمرين 1 | شرح الكود
اشرح هذا الكود سطراً سطراً: $stmt = $pdo->prepare("SELECT * FROM users WHERE email = :email"); $stmt->execute([':email' => $_POST['email']]); $users = $stmt->fetchAll();
السطرالشرح
$pdo->prepare(...) تحضير الاستعلام — يُفصل SQL عن البيانات. :email هو placeholder (مكان مؤقت آمن)
$stmt->execute([...]) تنفيذ الاستعلام مع البيانات الحقيقية بشكل آمن — يمنع SQL Injection
$stmt->fetchAll() جلب كل النتائج كـ Array PHP لنستطيع عرضها لاحقاً
PDO — تمرين 2 | لماذا prepare()؟
لماذا نستخدم prepare() بدلاً من كتابة الاستعلام مباشرةً؟
🔐 prepare() يفصل بين بنية الاستعلام SQL وبين البيانات المُدخلة.

بدون prepare: لو كتب المهاجم في حقل الإيميل ' OR '1'='1، يصبح الاستعلام خطيراً.

مع prepare: البيانات دائماً تُعامَل كنص عادي وليس كـ SQL — مهما كتب المهاجم، لا يُنفَّذ.
